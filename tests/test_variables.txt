# BASIC ADDITIONS AND SUBTRACTIONS
3x + 2 = 8,                                                           x = 2
x + 3 - 2 = 2x - 2,                                                   x = 3
6x - 22 - 3x = -2x - 2,                                               x = 4
x + 3 - 2 = 2x - 2,                                                   x = 3
x + 3 = 3,                                                            x = 0
x + 3 = -x - 1,                                                       x = -2
-2x + -3 = -x - 1,                                                    x = -2

# BASIC MULTIPLICATIONS
5x = 5 * 3,                                                           x = 3
5x = 2x,                                                              x = 0
5x = 5 * 3x,                                                          x = 0
3 * x = 3x,                                                           x = True for all x
5x = 5 * 3x - 20,                                                     x = 2

# CUSTOM VARIABLE
n + 2 = -n,                                                           n = -1

# BASIC POWERS
5x = 25^2,                                                            x = 125
15x = 30^2,                                                           x = 60

# FRACTIONS
5x/4 + 2 = 7,                                                         x = 4
3x/4 + 1/2 = 2,                                                       x = 2

# PARENTHESES
x + (5 * 3) = 10,                                                     x = -5
x + (5 * 3x) = -10,                                                   x = -5/8

# EQUATION WITH NON-INTEGER (FRACTION) RESULT
4x = -3,                                                              x = -3/4
5x/4 + 2/3 = 7 - x,                                                   x = 76/27

# CASE: PRODUCT OF PG AND GROUP IS 0
(x + 3)(x - 1) = 0,                                                   x = {-3, 1}
(x + 3)(2x - 1) = 0,                                                  x = {-3, 1/2}
2 - 2 = (x + 3)(2x - 1),                                              x = {-3, 1/2}
0 = (x + 3)(2x - 1)(x - 4),                                           x = {-3, 1/2, 4}
2x * (x - 4) = 0,                                                     x = {0, 4}
2(x - 4)(x + 3) = 0,                                                  x = {-3, 4}
2x/3 * (x - 4) = 0,                                                   x = {0, 4}
2/3(x - 4) = 0,                                                       x = {4}
(2x + 3)(4x + 3) + 6x - 6 = -(-6x + 6),                               x = {-3/2, -3/4}
18x/3 + (4x + 4)(x + 3) - 24/6 = -(-6x + 4),                          x = {-3, -1}

# ADVANCED USAGE
25x/10 + (2x + 3)(4x + 3) - (40/2 + 4)/(10 - 4) = -(-5x + 4) - 5x/2,  x = {-3/2, -3/4}

# NO SOLUTION
0x = 3,                                                               x = No Solution
3x - 4 = 3x + 2,                                                      x = No Solution
4x - x - 4 = 2x + x + 2,                                              x = No Solution
